    **Create TerminalPane.tsx** -- single pane wrapper with sentinel PTY spawn:
    Props: `{ paneId: string; ptyId: number }`

    Structure:
    ```typescript
    export function TerminalPane({ paneId, ptyId }: { paneId: string; ptyId: number }) {
      const containerRef = useRef<HTMLDivElement>(null);
      const focusedPaneId = usePaneTreeStore((s) => s.focusedPaneId);
      const setFocus = usePaneTreeStore((s) => s.setFocus);
      const setPtyId = usePaneTreeStore((s) => s.setPtyId);
      const isFocused = focusedPaneId === paneId;
      const [cwd, setCwd] = useState('');
      const [realPtyId, setRealPtyId] = useState(ptyId > 0 ? ptyId : null);

      // Sentinel PTY spawn: if ptyId is -1, spawn a new PTY on mount
      useEffect(() => {
        if (ptyId > 0) return; // Already has a real PTY
        if (realPtyId !== null) return; // Already spawning/spawned
        let cancelled = false;
        spawnTerminal((data) => {
          // onOutput callback -- will be wired to xterm by useTerminal
          // This needs careful handling: store the listener ref
        }).then((id) => {
          if (cancelled) return;
          setRealPtyId(id);
          setPtyId(paneId, id);
        });
        return () => { cancelled = true; };
      }, [ptyId, realPtyId, paneId, setPtyId]);

      // Only render terminal when we have a real PTY ID
      if (realPtyId === null) {
        return (
          <div className="h-full w-full flex items-center justify-center text-gray-500 text-sm">
            Starting terminal...
          </div>
        );
      }

      return (
        <div
          className="relative h-full w-full"
          style={{
            border: isFocused ? '1px solid #3b82f6' : '1px solid transparent',
            transition: 'border-color 150ms ease',
          }}
          onPointerDown={() => setFocus(paneId)}
        >
          <PaneHeader cwd={cwd} isFocused={isFocused} />
          <div ref={containerRef} className="h-full w-full" />
        </div>
      );
    }
    ```

    NOTE: The sentinel spawn pattern above is a guide. The executor must handle the `onOutput` callback properly -- `spawnTerminal` takes a callback that must forward data to xterm. Since xterm is created by `useTerminal` which only mounts when `realPtyId` is set, the callback should buffer or use a ref. The simplest approach: have `useTerminal` re-register the output listener after mount via the Tauri event system (which the existing implementation already uses via `listen('pty-output-{ptyId}', ...)`). If the current implementation uses event-based output (not callback), the `spawnTerminal` callback is just a no-op placeholder and the real data flows through Tauri events.

    Import xterm.css in TerminalPane.tsx (move from old Terminal.tsx): `import '@xterm/xterm/css/xterm.css';`
